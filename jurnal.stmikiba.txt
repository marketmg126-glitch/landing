    <script>
    function updateCountdown() {
    const wib = new Date(new Date().toLocaleString("en-US", {timeZone: "Asia/Jakarta"}));
    const h = wib.getHours(), m = wib.getMinutes(), s = wib.getSeconds();
    const totalSec = h*3600 + m*60 + s;
    const day = wib.getDay();

    // TOTO MACAU 6X
    const macauTimes = [{h:0,m:1},{h:13,m:0},{h:16,m:0},{h:19,m:0},{h:22,m:0},{h:23,m:0}];
    let nextMacau = 86400;
    macauTimes.forEach(t => {
        let timeSec = t.h*3600 + t.m*60;
        if (totalSec > timeSec) timeSec += 86400;
        const diff = timeSec - totalSec;
        if (diff < nextMacau) nextMacau = diff;
    });
    document.getElementById('macau').textContent = format(nextMacau);

    // HK 23:15
    document.getElementById('hk').textContent = format(calcDaily(23,15));

    // Sydney 14:05
    document.getElementById('sydney').textContent = format(calcDaily(14,5));

    // Singapore 17:45 (Selasa & Jumat libur)
    if (day === 2 || day === 5) {
        document.getElementById('sgp').textContent = "LIBUR";
    } else {
        document.getElementById('sgp').textContent = format(calcDaily(17,45));
    }

    // Kingkong 17:00 & 23:30
    const kk1 = 17*3600;
    const kk2 = 23*3600 + 30*60;
    let nextKK = totalSec < kk1 ? kk1 : (totalSec < kk2 ? kk2 : kk1 + 86400);
    if (totalSec > kk2) nextKK += 86400;
    document.getElementById('kingkong').textContent = format(nextKK - totalSec);

    function calcDaily(rh, rm) {
        let resultSec = rh*3600 + rm*60;
        let diff = resultSec - totalSec;
        if (diff < 0) diff += 86400;
        return diff;
    }

    function format(sec) {
        if (sec <= 0) return "00:00:00";
        const h = String(Math.floor(sec/3600)).padStart(2,'0');
        const m = String(Math.floor((sec%3600)/60)).padStart(2,'0');
        const s = String(sec%60).padStart(2,'0');
        return `${h}:${m}:${s}`;
    }
    }
    setInterval(updateCountdown, 1000);
    updateCountdown();
    </script>
